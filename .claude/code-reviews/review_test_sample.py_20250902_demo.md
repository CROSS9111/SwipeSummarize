```python
from typing import List, Dict, Union

def calculate_total(items: List[Dict[str, Union[str, int]]]) -> int:
    """
    Calculate the total price of all items.
    
    Args:
        items: List of item dictionaries with 'name' and 'price' keys
        
    Returns:
        Total price as integer
        
    Raises:
        TypeError: If items is not a list
        ValueError: If any item is missing the 'price' field
    """
    if not items:
        return 0
    
    if not isinstance(items, list):
        raise TypeError("Items must be a list")
    
    total = 0
    for item in items:
        if 'price' not in item:
            raise ValueError(f"Item {item} missing 'price' field")
        total += item['price']
    return total
```
